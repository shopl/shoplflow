## Usage

```tsx
import { Input } from "@shoplflow/base";

const Example = () => {

    const [value, setValue] = useState('')
    const nowLength = value?.length ?? 0;

    return (
        <Input 
            onDelete={() => {
                setValue('')
            }}
            nowLength={nowLength}   
        />
    )
}
```
